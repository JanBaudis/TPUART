28 
29  // Using the USART0 of Port C - so set the IO Pins as in- and output.
30  PORTC.DIRCLR = PIN2_bm; // Sets PC2 (RXD0) as input.
31  PORTC.DIRSET = PIN3_bm; // Sets PC3 (TXD0) as output.
32 
33  /* Use USARTC0 and initialize buffers. */
34  USART_InterruptDriver_Initialize(&USART_DATA_TP, &USART_TP, USART_DREINTLVL_LO_gc);
35 
36  /* USARTC0, 8 Data bits, Even Parity, 1 Stop bit. */
37  USART_Format_Set(USART_DATA_TP.usart, USART_CHSIZE_8BIT_gc,USART_PMODE_EVEN_gc, false); //TPUART1 Datasheet(p.10) is missing the parity specs - TPUART2 DS(p.21) says even, so lets try even
38 
39  /* Enable RXC interrupt. */
40  USART_RxdInterruptLevel_Set(USART_DATA_TP.usart, USART_RXCINTLVL_LO_gc);
41 
42  /* Set Baudrate to 19200 bps:
43  * Use the 32MHz clock.
44  * If ScaleFactor < 0 BSEL = (1/(2^(ScaleFactor)*16))*(((I/O clock frequency)/Baudrate)-1) is documented in the avr1307-docs but the excel sheet from atmel says 1651
45  */
46  USART_Baudrate_Set(&USART_TP, 1651 , 0x0C); // The third param uses uint8 - little bit confusing but as long as the 4bits for the BSCALE are set/coded correct (two`s complement)-7(1001) to +7(0111) - so i think its better to use the hexadecimal way
47 
48  /* Enable both RX and TX. */
49  USART_Rx_Enable(USART_DATA_TP.usart);
50  USART_Tx_Enable(USART_DATA_TP.usart);
51 
52  /* Enable PMIC interrupt level low. */
53  PMIC.CTRL |= PMIC_LOLVLEX_bm;
54 
55 }
